##第一种方法、通过递归解析解决
		//深复制，要想达到深复制就需要用递归
function deepCopy(o, c) {
    var c = c || {}
    for (var i in o) {
        if (typeof o[i] === 'object') {//要考虑深复制问题了
            if (o[i].constructor === Array) {//这是数组
                c[i] = []
            } else {//这是对象
                c[i] = {}
            }
            deepCopy(o[i], c[i])
        } else {
            c[i] = o[i]
        }
    }
    return c
}
